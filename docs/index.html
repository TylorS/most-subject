Examples

import {holdSubject} from 'most-subject'

const stream = holdSubject(3)

stream.next(1)
stream.next(2)
stream.next(3)

stream.map(fn).observe(x => console.log(x))
// 1
// 2
// 3

setTimeout(() => stream.complete(), 10)
